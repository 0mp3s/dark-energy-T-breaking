\section{Context: Paper~1 --- Secluded Majorana SIDM}
%=====================================================================

The companion paper~\cite{paper1} establishes a \emph{secluded} Majorana dark matter model with a light scalar mediator $\phi$:
\begin{equation}
\mathcal{L}_{\mathrm{SIDM}} = \tfrac{1}{2}\bar\chi(i\!\not\!\partial - m_\chi)\chi + \tfrac{1}{2}(\partial\phi)^2 - V_0(\phi) - \tfrac{1}{2}\bar\chi\bigl(y_s + iy_p\gamma^5\bigr)\chi\,\phi
\end{equation}
with three free parameters $(m_\chi,\,m_\phi,\,\alpha_s)$ determining self-interactions. The Variable Phase Method (VPM) solver, Majorana spin statistics ($w_\ell^{\mathrm{even}}=1$, $w_\ell^{\mathrm{odd}}=3$), and Boltzmann relic density are all implemented and validated. Key results:
\begin{itemize}[nosep]
  \item 80{,}142 viable points from a 2.8M-point scan; 17 relic-constrained benchmarks with $\Omega_\chi h^2=0.120$ and $\chi^2/\nu<0.68$ for 13 astrophysical systems.
  \item MAP benchmark: $m_\chi=98.19\GeV$, $m_\phi=9.66\MeV$, $\alpha_D=3.274\times10^{-3}$.
  \item Majorana fingerprint: $\sigma_T^{\mathrm{Maj}}/\sigma_T^{\mathrm{Dir}}=2.2$ at $v\approx60$~km/s (triplet dominance 82\%).
  \item Fully secluded: $\sigma_{\mathrm{SI}}=\sigma_{\mathrm{SD}}=0$ (Higgs portal excluded by $3\times10^4$).
\end{itemize}
Published data and code: Zenodo DOI \texttt{10.5281/zenodo.19225823}.

\section{Extension: Dark Energy from the Same Lagrangian}
%=====================================================================

\subsection{Core idea}
We promote the CP phase $\theta$ in the Yukawa coupling to a dynamical field $\sigma$ (a dark axion / dark pion):
\begin{equation}\label{eq:promotion}
  y_s + iy_p\gamma^5 \;\longrightarrow\; y\,e^{i\gamma^5\sigma/f}
  \;=\; y\cos\!\Bigl(\frac{\sigma}{f}\Bigr) + iy\sin\!\Bigl(\frac{\sigma}{f}\Bigr)\gamma^5
\end{equation}
where $f$ is the decay constant, and $\sigma$ acquires a potential from dark QCD chiral symmetry breaking:
\begin{equation}\label{eq:potential}
  \boxed{V(\sigma) = \Ld^4\Bigl(1 - \cos\frac{\sigma}{f}\Bigr)}
\end{equation}
with $\Ld$ the dark QCD confinement scale and $m_\sigma = \Ld^2/f$ the dark pion mass.

\subsection{Dark QCD sector}
A confining dark gauge group SU(2)$_d$ with 3 Majorana quarks produces the dark pion $\sigma$.
The confinement scale is set by dimensional transmutation:
\begin{equation}\label{eq:transmutation}
  \Ld = \mu\cdot\exp\!\Bigl(-\frac{2\pi}{b_0\,\alpha_d(\mu)}\Bigr),
  \qquad b_0 = \tfrac{19}{3}\;\text{(SU(2)$_d$, 3 Majorana)}
\end{equation}
With $\alpha_d(m_\chi)\approx0.032$ and $\mu=m_\chi=98\GeV$, this yields $\Ld\approx2\meV$ ---
the dark energy scale --- \emph{without} fine-tuning. The coupling $\alpha_d\sim 0.03$ is comparable to $\alpha_{\mathrm{EM}}$.

\subsection{Cosmological misalignment}
At reheating, $\sigma$ sits at a misalignment angle $\theta_i \equiv \sigma_i/f \sim 3$~rad (near the CP-preserving hilltop at $\theta=\pi$). While $H\gg m_\sigma$, the field is frozen by Hubble friction. Today $V(\sigma_{\mathrm{today}})\approx\rho_\Lambda$.

\section{The Coupled ODE: $\sigma(t)\otimes$Friedmann $\to H_0$}
%=====================================================================

We solve a coupled ODE system in $N = \ln a$ (e-fold time):
\begin{align}
  \frac{d\sigma}{dN} &= p, \\
  \frac{dp}{dN} &= -(3-\varepsilon)\,p - \frac{V'(\sigma)}{H^2}, \\
  H^2 &= \frac{\rho_r + \rho_m + V(\sigma)}{3\Mpl^2 - \tfrac{1}{2}p^2}, \\
  \varepsilon &= \frac{\tfrac{4}{3}\rho_r + \rho_m + H^2 p^2}{2\Mpl^2 H^2}
\end{align}
where $\rho_r\propto a^{-4}$, $\rho_m = \rho_\chi(a) + \rho_b(a) \propto a^{-3}$.

\medskip
\noindent
\textbf{Inputs} (6 Lagrangian parameters):
\[
  \underbrace{m_\chi,\;m_\phi,\;\alpha_D}_{\text{SIDM sector}\;\to\;\Omega_\chi h^2}
  \;,\qquad
  \underbrace{f,\;\Ld,\;\theta_i}_{\text{dark QCD}\;\to\;V(\sigma)}
\]

\noindent
\textbf{Output}: $H_0$ [km/s/Mpc] --- derived, not input.
